Ejercicio 2 - Validador de Fechas

from datetime import datetime

# Enunciado: formato dd/mm/yy
cadena_validar = ["13/02/25", "hola carmela", "12 34 56", "14/06/2026", "bra, bra", "56/13/26"]

fechas_correctas = []
fechas_incorrectas = []

for cadena in cadena_validar:
    try:
        # Solo validamos formato dd/mm/yy seg√∫n enunciado
        datetime.strptime(cadena, "%d/%m/%y")
        fechas_correctas.append(cadena)
    except:
        fechas_incorrectas.append(cadena)

# Mostrar en consola como pide el enunciado
print("Fechas correctas:")
for fecha in fechas_correctas:
    print(f'"{fecha}"')

print("\nFechas incorrectas:")
for fecha in fechas_incorrectas:
    print(f'"{fecha}"')

Ejercicio 3 - Separar Fechas Anteriores y Posteriores a Hoy

# EJERCICIO 3
from datetime import datetime

# Usamos solo las fechas correctas del ejercicio anterior
fechas_anteriores = []
fechas_posteriores = []

for fecha_str in fechas_correctas:
    # Convertir a objeto datetime
    fecha_obj = datetime.strptime(fecha_str, "%d/%m/%y")
    
    if datetime.today() > fecha_obj:
        fechas_anteriores.append(fecha_str)
    else:
        fechas_posteriores.append(fecha_str)

print("Fechas anteriores a hoy:")
for fecha in fechas_anteriores:
    print(f'"{fecha}"')

print("\nFechas posteriores a hoy:")
for fecha in fechas_posteriores:
    print(f'"{fecha}"')
