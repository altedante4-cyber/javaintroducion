Formatear y Parsear Fechas

Convertir fechas a texto (strftime) y texto a fechas (strptime).

Python 
from datetime import datetime

# strftime: datetime -> string
ahora = datetime.now()
print(ahora.strftime("%Y-%m-%d"))           # 2024-01-15
print(ahora.strftime("%d/%m/%Y"))           # 15/01/2024
print(ahora.strftime("%H:%M:%S"))           # 14:30:45
print(ahora.strftime("%I:%M %p"))           # 02:30 PM
print(ahora.strftime("%A, %d %B %Y"))       # lunes, 15 enero 2024

# strptime: string -> datetime
fecha_str = "2024-12-25 14:30:00"
fecha_dt = datetime.strptime(fecha_str, "%Y-%m-%d %H:%M:%S")

# C√≥digos comunes:
# %Y = a√±o (2024), %y = a√±o (24)
# %m = mes (01-12), %B = mes nombre, %b = mes abreviado
# %d = d√≠a (01-31)
# %H = hora 24h, %I = hora 12h, %M = minutos, %S = segundos
# %p = AM/PM

# D√≠as h√°biles
def dias_habiles(inicio, fin):
    dias = 0
    actual = inicio
    while actual <= fin:
        if actual.weekday() < 5:  # 0-4 = lunes a viernes
            dias += 1
        actual += timedelta(days=1)
    return dias

from datetime import date

def calcular_edad(fecha_nacimiento):
    hoy = date.today()
    anos = hoy.year - fecha_nacimiento.year
    if (hoy.month, hoy.day) < (fecha_nacimiento.month, fecha_nacimiento.day):
        anos -= 1
    return anos

# Uso

@lru_cache(maxsize=None)

# Suma anidada
def suma_anidada(objeto):
    if isinstance(objeto, (int, float)):
        return objeto
    if isinstance(objeto, (list, tuple)):
        return sum(suma_anidada(item) for item in objeto)
    if isinstance(objeto, dict):
        return sum(suma_anidada(v) for v in objeto.values())
    return 0

# B√∫squeda binaria
def busqueda_binaria(lista, objetivo, inicio=0, fin=None):
    if fin is None:
        fin = len(lista) - 1
    if inicio > fin:
        return -1
    
    medio = (inicio + fin) // 2
    if lista[medio] == objetivo:
        return medio
    elif lista[medio] < objetivo:
        return busqueda_binaria(lista, objetivo, medio + 1, fin)
    else:
        return busqueda_binaria(lista, objetivo, inicio, medio - 1)
